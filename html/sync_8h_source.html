37 static inline bool __always_inline
38 cpu_sync_mutex(cpu_sync_lock_t b)
39 {
40  register uint8_t *bios_address;
41  bios_address = (uint8_t *)HWRAM_UNCACHED(0x00000B00);
42 
43  register uint32_t result;
44  result = 0;
45 
46  /* Currently using cpu_sync_mutex() to compare the value results in
47  * extra instructions being emitted. Because TAS.B sets the T bit in SR,
48  * there shouldn't be a need for the MOVT instruction */
49 
50  __asm__ volatile ("\tadd %[b], %[bios_address]\n"
51  "\ttas.b @%[out]\n"
52  "\tmovt %[result]"
53  /* Output */
54  : [out] "=r" (bios_address),
55  [result] "=r" (result)
56  /* Input */
57  : [bios_address] "r" (bios_address),
58  [b] "r" (b));
59 
60  return result;
61 }
