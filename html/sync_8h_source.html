41 static inline bool __always_inline
42 cpu_sync_mutex(cpu_sync_lock_t b)
43 {
44  register uint8_t *bios_address;
45  bios_address = (uint8_t *)HWRAM_UNCACHED(0x00000B00);
46 
47  register uint32_t result;
48  result = 0;
49 
50  /* Currently using cpu_sync_mutex() to compare the value results in
51  * extra instructions being emitted. Because TAS.B sets the T bit in SR,
52  * there shouldn't be a need for the MOVT instruction */
53 
54  __asm__ volatile ("\tadd %[b], %[bios_address]\n"
55  "\ttas.b @%[out]\n"
56  "\tmovt %[result]"
57  /* Output */
58  : [out] "=r" (bios_address),
59  [result] "=r" (result)
60  /* Input */
61  : [bios_address] "r" (bios_address),
62  [b] "r" (b));
63 
64  return result;
65 }
