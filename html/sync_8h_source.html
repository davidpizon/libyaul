24 static inline bool __always_inline
25 cpu_sync_mutex(uint8_t b)
26 {
27  register uint8_t *bios_address;
28  bios_address = (uint8_t *)HWRAM_UNCACHED(0x00000B00);
29 
30  register uint32_t result;
31  result = 0;
32 
33  /* Currently using cpu_sync_mutex() to compare the value results in
34  * extra instructions being emitted. Because TAS.B sets the T bit in SR,
35  * there shouldn't be a need for the MOVT instruction */
36 
37  __asm__ volatile ("\tadd %[b], %[bios_address]\n"
38  "\ttas.b @%[out]\n"
39  "\tmovt %[result]"
40  /* Output */
41  : [out] "=r" (bios_address),
42  [result] "=r" (result)
43  /* Input */
44  : [bios_address] "r" (bios_address),
45  [b] "r" (b));
46 
47  return result;
48 }
