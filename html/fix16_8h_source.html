150 static inline fix16_t __always_inline
151 fix16_mul(const fix16_t a, const fix16_t b)
152 {
153  __register uint32_t mach;
154  __register fix16_t out;
155 
156  __asm__ volatile ("\tdmuls.l %[a], %[b]\n"
157  "\tsts mach, %[mach]\n"
158  "\tsts macl, %[out]\n"
159  "\txtrct %[mach], %[out]\n"
160  /* Output */
161  : [mach] "=&r" (mach),
162  [out] "=&r" (out)
163  /* Input */
164  : [a] "r" (a),
165  [b] "r" (b)
166  : "mach", "macl");
167 
168  return out;
169 }
