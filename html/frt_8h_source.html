134 static inline void __always_inline
135 cpu_frt_interrupt_priority_set(uint8_t priority)
136 {
137  /* Set the interrupt priority level for FRT (shared amongst all FRT
138  * related interrupts */
139  MEMORY_WRITE_AND(16, CPU(IPRB), 0xF7FF);
140  MEMORY_WRITE_OR(16, CPU(IPRB), (priority & 0x0F) << 8);
141 }
