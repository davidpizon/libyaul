64 static inline void __always_inline
65 smpc_smc_wait(bool enter)
66 {
67  volatile uint8_t *reg_sf;
68  reg_sf = (volatile uint8_t *)SMPC(SF);
69 
70  while ((*reg_sf & 0x01) == 0x01);
71 
72  if (enter) {
73  *reg_sf = 0x01;
74  }
