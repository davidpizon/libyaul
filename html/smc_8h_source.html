73 static inline void __always_inline
74 smpc_smc_wait(bool enter)
75 {
76  volatile uint8_t *reg_sf;
77  reg_sf = (volatile uint8_t *)SMPC(SF);
78 
79  while ((*reg_sf & 0x01) == 0x01);
80 
81  if (enter) {

147 static inline uint8_t __always_inline
148 smpc_smc_intback_call(uint8_t ireg0, uint8_t ireg1)
149 {
150  /* Fetch peripheral data "SMPC" status */
151  smpc_smc_wait(true);
152 
153  MEMORY_WRITE(8, IREG(0), ireg0 & 0x01);
154  MEMORY_WRITE(8, IREG(1), ireg1 & 0x7F);
155  MEMORY_WRITE(8, IREG(2), 0xF0);
156 
157  MEMORY_WRITE(8, SMPC(COMREG), SMPC_SMC_INTBACK);
158 
159  return 0;
160 }
